Esercizio 1

Preparare un programma che calcoli le soluzioni dell'equazione di 2° grado ax2+bx+c=0.

#include <stdio.h>
#include <math.h>
void main()
{
float a,b,c;
printf("a=");scanf("%f",&a);
printf("b=");scanf("%f",&b);
printf("c=");scanf("%f",&c);
float delta;
delta=pow(b,2)-4*a*c;
if(delta>=0)
{if(delta==0)
printf("due soluzioni concidenti x1=x2=%.2f",-b/(2*a));
else
printf("due soluzioni x1=%.2f x2=%.2f",(-b+sqrt(delta))/(2*a),(-b-sqrt(delta))/(2*a));
}
else
printf("Nessuna soluzione");
}
