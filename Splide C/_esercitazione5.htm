{

printf("b=");scanf("%d",&b);
printf("MCD tra %d ed %d = %d",a,b,MCD(a,b));
}
int MCD(int a,int b)
{int max,min;
max = (a>b) ? a : b;
min= (a<b) ? a : b;
int r1=max%min;
if(r1==0) return(min);
return(MCD(min,r1));
}
